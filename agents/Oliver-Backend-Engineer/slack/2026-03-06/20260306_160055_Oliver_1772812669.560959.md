## Database Design for Venture Capital

### Tables
- ventures
  - id UUID PRIMARY KEY
  - slug TEXT UNIQUE
  - name TEXT NOT NULL
  - short_description TEXT
  - description TEXT
  - status venture_status_enum NOT NULL DEFAULT 'prospective'  -- e.g. prospective, active, acquired, closed
  - stage venture_stage_enum NOT NULL DEFAULT 'seed'            -- e.g. pre-seed, seed, series_a...
  - headquarters TEXT
  - website TEXT
  - founded_at DATE
  - is_confidential BOOLEAN DEFAULT FALSE
  - owner_id UUID REFERENCES users(id)  -- product owner / record owner
  - metadata JSONB DEFAULT '{}'         -- flexible key/value (market, tags, etc.)
  - search_vector TSVECTOR GENERATED ALWAYS AS (
      setweight(to_tsvector('english', coalesce(name, '')), 'A') ||
      setweight(to_tsvector('english', coalesce(short_description, '')), 'B') ||
      setweight(to_tsvector('english', coalesce(description, '')), 'C')
    ) STORED
  - created_at TIMESTAMP WITH TIME ZONE DEFAULT now()
  - updated_at TIMESTAMP WITH TIME ZONE DEFAULT now()
  - deleted_at TIMESTAMP WITH TIME ZONE NULL

- venture_founders
  - id UUID PRIMARY KEY
  - venture_id UUID NOT NULL REFERENCES ventures(id) ON DELETE CASCADE
  - user_id UUID NULL REFERENCES users(id)                   -- optional link if founder is a user
  - name TEXT NOT NULL
  - email BYTEA NULL                                        -- encrypted (See privacy notes)
  - role TEXT
  - bio TEXT
  - links JSONB DEFAULT '{}'                                -- linkedin, twitter, etc.
  - created_at TIMESTAMP WITH TIME ZONE DEFAULT now()
